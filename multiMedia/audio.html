      $(function(){
        $("#jquery_jplayer_1").jPlayer({
          ready: function (event) {
            $(this).jPlayer("setMedia", {
              title: "",
              mp3: "dream.mp3",
            });
          },
          swfPath: "../../lib/jqueryPlugin/jPlayer-2.9.1/dist/jplayer",
          supplied:"mp3",
          wmode: "window",
          useStateClassSkin: true,
          autoBlur: false,
          smoothPlayBar: true,
          keyEnabled: true,
          remainingDuration: true,
          toggleDuration: true
        });

      });
