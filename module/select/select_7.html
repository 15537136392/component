<input type="text" id="J_input" readonly />


$("#J_select").NiceSelect({
	url:"../../../data/selectDatas.json",
	datas:"codeType=test",
	id:"testId",
	name:"testId",
	defaultValue:"50",
	allJson:true, 		//为true，则会将每条json添加至对应的option节点上
	handler:function(){
		var txt = $(this).find('option:checked').attr('alljson');
		txt = JSON.parse(txt);
		$('#J_input').val(txt.value);
	}
});
