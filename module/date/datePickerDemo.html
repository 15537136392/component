javascript部分 

      <script type="text/javascript">
          $(document).ready(function(){
              //1、简单使用日期控件
              $('#date1').on('focus',WdatePicker);//1、正常日期选择绑定事件
              //2、日期选择-限制最小日期
              $('#date2').on('focus',function(){
                  var dateParam={
                      minDate:'%y-%M-%d'//选择的最小值是今天;某一天的格式 例如:2015-08-20
                  }
                  WdatePicker(dateParam);
              })
              $('#date3').on('focus',function(){
                  var dateParam={
                     minDate:'%y-%M-#{%d-1}'//动态时间运算#{}
                  }
                  WdatePicker(dateParam);
              });
              //3、日期选择-限制最大日期
              $('#date4').on('focus',function(){
                  var dateParam={
                      maxDate:'%y-%M-%d'//选择的最大值是今天;某一天的格式 例如:2015-08-20
                  }
                  WdatePicker(dateParam);
              })
              $('#date5').on('focus',function(){
                  var dateParam={
                     maxDate:'%y-%M-#{%d+1}'//动态时间运算#{}
                  }
                  WdatePicker(dateParam);
              });
                //4、日期起止范围
               $('#date6').on('focus',function(){
                  var dateParam={
                     maxDate:'#F{$dp.$D(\'date7\')}'//最大不能大于ID为date7的日期控件的值
                  }
                  WdatePicker(dateParam);
              });
              $('#date7').on('focus',function(){
                  var dateParam={
                     minDate:'#F{$dp.$D(\'date6\')}'//最小不能小于ID为date6的日期控件的值
                  }
                  WdatePicker(dateParam);
              });
              $('#date8').on('focus',function(){
                  var dateParam={
                     maxDate:'#F{$dp.$D(\'date9\',{d:-3})}'//最大时间选择date9日期的前三天
                  }
                  WdatePicker(dateParam);
              });
              $('#date9').on('focus',function(){
                  var dateParam={
                     minDate:'#F{$dp.$D(\'date8\',{d:3})}'//最小时间选择date8的后三天
                  }
                  WdatePicker(dateParam);
              });
              //5、日期格式
              $('#date10').on('focus',function(){
                  var dateParam={
                    dateFmt:'yyyy-MM-dd HH:mm:ss'//日期格式yyyy-MM-dd HH:mm:ss
                  }
                  WdatePicker(dateParam);
              });
              $('#date11').on('focus',function(){
                  var dateParam={
                     dateFmt:'HH:mm:ss'//日期格式HH:mm:ss
                  }
                  WdatePicker(dateParam);
              });
              $('#date12').on('focus',function(){
                  var dateParam={
                     dateFmt:'yyyy-MM-dd'//日期格式yyyy-MM-dd
                  }
                  WdatePicker(dateParam);
              });
              //6、自动纠错功能(不提示直接纠错)
              $('#date13').on('focus',function(){
                  var dateParam={
                     dateFmt:'yyyy-MM-dd',//日期格式yyyy-MM-dd
                     errDealMode:1//errDealMode为1时输入错误自动恢复前一次正确
                  }
                  WdatePicker(dateParam);
              });
          })
</script>
